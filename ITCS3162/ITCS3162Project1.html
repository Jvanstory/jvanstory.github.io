import pandas as pd
import matplotlib.pyplot as plt

# Load your weather data into a DataFrame (replace with your actual data)
df = pd.read_csv(r'C:\Users\joeva\Downloads\WeatherEvents_Jan2016-Dec2022.csv')
columns_to_drop = ['EndTime(UTC)', 'TimeZone', 'AirportCode', 'LocationLat', 'LocationLng', 'City', 'County', 'ZipCode']
df.drop(columns=columns_to_drop, inplace=True)
df.dropna(inplace=True)

# Convert the 'StartTime(UTC)' column to a datetime object
df['StartTime(UTC)'] = pd.to_datetime(df['StartTime(UTC)'])

# Filter the DataFrame to exclude 'UNK' severity
df = df[df['Severity'] != 'UNK']

# Group the data by month and count the number of events in each month
monthly_severity = df.groupby(df['StartTime(UTC)'].dt.to_period('M'))['Severity'].count()

# Convert time period objects to string representations
monthly_severity.index = monthly_severity.index.strftime('%Y-%m')

# Create a new figure for the line graph of monthly severity
plt.figure(figsize=(12, 6))
plt.plot(monthly_severity.index, monthly_severity.values, marker='o', linestyle='-', color='b')
plt.title('Monthly Severity of Weather Events')
plt.xlabel('Time (Year-Month)')
plt.ylabel('Number of Events')
plt.grid(True, linestyle='--', alpha=0.7)
plt.xticks(monthly_severity.index[::12], rotation=45, fontsize=10)  # Label every 12th month

# Display the figure
plt.show()

# Extract the year from the datetime
df['Year'] = df['StartTime(UTC)'].dt.year

# Group the data by year and count the number of events in each year
yearly_event_counts = df.groupby('Year')['Severity'].count()

# Create a bar plot to visualize the total number of events per year
plt.figure(figsize=(12, 6))
plt.bar(yearly_event_counts.index, yearly_event_counts.values, color='blue')
plt.title('Total Number of Weather Events per Year')
plt.xlabel('Year')
plt.ylabel('Number of Events')
plt.xticks(rotation=45)
plt.grid(True, linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()

# Create a new figure for the stacked bar plot
plt.figure(figsize=(12, 6))

# Create a pivot table to count severity levels by year
df['Year'] = df['StartTime(UTC)'].dt.year
pivot_table = df.pivot_table(index='Year', columns='Severity', aggfunc='size', fill_value=0)

# Create a stacked bar plot
pivot_table.plot(kind='bar', stacked=True, ax=plt.gca())
plt.title('Stacked Bar Plot of Weather Event Severity Over Years')
plt.xlabel('Year')
plt.ylabel('Number of Events')
plt.xticks(rotation=45)

# Display the figure
plt.show()

# Create a new figure for the line graph of severe events per year
plt.figure(figsize=(12, 6))
severe_events_per_year = df[df['Severity'] == 'Severe'].groupby(df['StartTime(UTC)'].dt.year)['Severity'].count()
plt.plot(severe_events_per_year.index, severe_events_per_year.values, marker='o', linestyle='-', color='r')
plt.title('Number of Severe Weather Events per Year')
plt.xlabel('Year')
plt.ylabel('Number of Severe Events')
plt.grid(True, linestyle='--', alpha=0.7)

# Display the figure
plt.show()
